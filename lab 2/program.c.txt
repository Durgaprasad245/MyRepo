#include <stdio.h>

void insertionSort(int arr[], int N) {
    int temp, currLoc;
    for (int i = 1; i < N; i++) {
        currLoc = i;
        while (currLoc > 0 && arr[currLoc - 1] > arr[currLoc]) {
            temp = arr[currLoc];
            arr[currLoc] = arr[currLoc - 1];
            arr[currLoc - 1] = temp;
            currLoc--;
        }
    }
}

int main() {
    int N;
    printf("Please enter the number of elements in the array: ");
    scanf("%d", &N);

    int arr[N];
    for (int i = 0; i < N; i++) {
        printf("Please enter element %d of the array: ", i + 1);
        scanf("%d", &arr[i]);
    }

   
    printf("Given array is: [");
    for (int i = 0; i < N - 1; i++) {
        printf("%d, ", arr[i]);
    }
    printf("%d]\n", arr[N - 1]);

    
    insertionSort(arr, N);

    
    printf("Sorted array is: [");
    for (int i = 0; i < N - 1; i++) {
        printf("%d, ", arr[i]);
    }
    printf("%d]\n", arr[N - 1]);

    return 0;
}
