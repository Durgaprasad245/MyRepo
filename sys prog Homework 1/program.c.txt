#include <stdio.h>
#include <stdbool.h>
#include <math.h>


int isPrime(int num) {
    if (num <= 1) {
        return 0;  
    }
    for (int i = 2; i <= sqrt(num); ++i) {
        if (num % i == 0) {
            return 0;  
        }
    }
    return 1;  
}


int nearestPrimeBelow(int num) {
    for (int i = num - 1; i > 1; --i) {
        if (isPrime(i)) {
            return i;
        }
    }
    return 2;  
}


int sumOfDigits(int num) {
    int sum = 0;
    while (num > 0) {
        sum += num % 10;
        num /= 10;
    }
    return sum;
}


void numberTransformer(int n) {
    if (isPrime(n) && n != 3 && n != 5) {
        printf("Go Blazers\n");
    } else if ((int)log2(n) == log2(n)) {
        int nearestPrime = nearestPrimeBelow(n);
        int result = n + nearestPrime;
        printf("%d\n", result);
    } else if (sumOfDigits(n) % 3 == 0) {
        printf("CS\n");
    } else if (n % 3 == 0 && n % 5 == 0) {
        printf("UAB CS 332&532\n");
    } else {
        int squareResult = pow(sumOfDigits(n), 2);
        printf("%d\n", squareResult);
    }
}


bool UABNumber(int n2) {
    
    if (n2 < 0) {
        printf("Invalid input: Please enter a non-negative integer.\n");
        return false;
    }

    
    int sum = 0;
    for (int i = 1; i <= n2 / 2; ++i) {
        if (n2 % i == 0) {
            sum += i;
        }
    }

    
    return (sum == n2);
}


int reverseNum(int n3) {
    int reversed = 0;
    while (n3 > 0) {
        reversed = reversed * 10 + n3 % 10;
        n3 /= 10;
    }
    return reversed;
}


int smallerThanIndex(int numbers[5]) {
    int count = 0;
    for (int i = 0; i < 5; ++i) {
        if (numbers[i] < i) {
            count++;
        }
    }
    return count;
}

void arrayDetails(int array[6]) {
    int min = array[0], max = array[0], minIndex = 0, maxIndex = 0;
    double mean = array[0];

    for (int i = 1; i < 6; ++i) {
        if (array[i] < min) {
            min = array[i];
            minIndex = i;
        }

        if (array[i] > max) {
            max = array[i];
            maxIndex = i;
        }

        mean += array[i];
    }

    mean /= 6;

    printf("Minimum Value: %d\n", min);
    printf("Minimum Value's Index: %d\n", minIndex);
    printf("Mean: %.2lf\n", mean);  
    printf("Maximum Value: %d\n", max);
    printf("Maximum Value's Index: %d\n", maxIndex);
}

int main() {
    int n1, n2, n3;

    
    printf("Enter an integer for numberTransformer: ");
    scanf("%d", &n1);
    numberTransformer(n1);

    
    printf("Enter an integer to check UABNumber: ");
    scanf("%d", &n2);
    printf("%s\n", UABNumber(n2) ? "true" : "false");

    
    printf("Enter a positive integer for reverseNum: ");
    scanf("%d", &n3);
    int reversed = reverseNum(n3);
    printf("Reversed Number: %d\n", reversed);

    
    int userNumbers[5];
    printf("Enter 5 integers for smallerThanIndex function:\n");
    for (int i = 0; i < 5; ++i) {
        scanf("%d", &userNumbers[i]);
    }
    int count = smallerThanIndex(userNumbers);
    printf("Count of Numbers Smaller Than Index: %d\n", count);

    
    int userArray[6];
    printf("Enter 6 integers for arrayDetails function:\n");
    for (int i = 0; i < 6; ++i) {
        scanf("%d", &userArray[i]);
    }
    arrayDetails(userArray);

    return 0;
}
